def create_dupfile_conllu(inputfile, outputfile, percentage=None):
    """
    Processes a CoNLL-U formatted dataset and duplicates sentences based on the number of predicates.
    Optionally truncates the dataset to keep a random subset of sentences.

    Parameters:
    inputfile (str): Path to a CoNLL-U formatted text file.
    outputfile (str): Path to a new file with duplicated sentences.
    percentage (float, optional): If provided, keeps only the specified percentage of sentences.
                                  Must be between 0 and 1. Default is None (keep all sentences).

    Ensures each predicate in the 11th column has its own instance of the sentence, keeping only the
    corresponding argument structure. The `# sent_id` and `# text` lines are retained for each duplicate.
    If a '# newdoc id' line appears, it is only written once at the start of its group.
    """
    
    with open(inputfile, 'r', encoding='utf-8') as file:
        lines = file.readlines()

    all_sentences = []
    current_sentence = []
    sent_metadata = []
    newdoc_id = None

    for line in lines:
        if line.startswith("# newdoc id"):
            newdoc_id = line.strip()
        elif line.startswith("#"):  # other metadata (# sent_id, # text)
            sent_metadata.append(line.strip())
        elif line.strip():  # token line
            current_sentence.append(line.strip().split("\t"))
        else:  # empty line (sentence boundary)
            if current_sentence:
                all_sentences.append((newdoc_id, sent_metadata, current_sentence))
            current_sentence = []
            sent_metadata = []
            newdoc_id = None

    # make sure that the last sentence is added if the file does not end with an empty line
    if current_sentence:
        all_sentences.append((newdoc_id, sent_metadata, current_sentence))

    # if subset is requested, do random sampling
    if percentage is not None:
        num_sentences = len(all_sentences)
        num_to_keep = max(1, int(num_sentences * percentage))
        all_sentences = random.sample(all_sentences, num_to_keep)

    output_lines = []

    # process each sentence and duplicate based on predicates
    for newdoc_id, sent_metadata, current_sentence in all_sentences:
        df = pd.DataFrame(current_sentence)
        df.columns = list(range(df.shape[1]))

        if df.shape[1] < 12:
            continue  # skip sentences without predicate/argument columns

        predicates = df[10].tolist()
        pred_indices = [i for i, p in enumerate(predicates) if p != '_']

        if not pred_indices:
            continue

        first_instance = True  # make sure '# newdoc id' appears only on the first version of the sentence

        for pred_idx in pred_indices:
            new_df = df.copy()

            # update predicate column, keep only the current predicate
            for i in range(len(new_df)):
                new_df.iloc[i, 10] = '_' if i != pred_idx else new_df.iloc[i, 10]

            arg_col = 11 + pred_indices.index(pred_idx)  # get correct argument column
            if arg_col < new_df.shape[1]:  
                new_df[11] = new_df.iloc[:, arg_col]
            else:
                new_df[11] = '_'

            # make sure to keep only relevant columns (first 12)
            new_df = new_df.iloc[:, :12]  

            # append metadata
            if first_instance and newdoc_id:  
                output_lines.append(newdoc_id)
            output_lines.extend(sent_metadata)
            first_instance = False

            # append the duplicated sentence
            output_lines.extend(["\t".join(map(str, row)) for row in new_df.values])
            output_lines.append("")  # blank line between sentences

    with open(outputfile, 'w', encoding='utf-8') as out_file:
        out_file.write("\n".join(output_lines))

def count_sentences_tokens(conllu_file):
    """
    Counts the number of sentences and tokens in a conllu file.

    Parameters:
    conllu_file: path to the input file

    Returns:
    num_sentences: number of sentences in the dataset
    num_tokens: total number of tokens in the dataset
    """
    num_sentences = 0
    num_tokens = 0

    with open(conllu_file, 'r', encoding='utf-8') as file:
        for line in file:
            line = line.strip()
            if line.startswith('# sent_id'):
                num_sentences += 1
            elif line and not line.startswith('#'):
                num_tokens += 1  # count tokens (non-empty, non-metadata lines)

    return num_sentences, num_tokens

def conllu_to_dataframe(filepath):
    rows = []
    current_sent_id = None
    sent_id_count = {}
    
    with open(filepath, 'r', encoding='utf-8') as file:
        for line in file:
            line = line.strip()
            # process comment lines to identify sentence ids
            if line.startswith("#"):
                if line.startswith("# sent_id"):
                    # extract the base sentence id
                    parts = line.split("=", 1)
                    base_id = parts[1].strip()
                    # add count for this base id
                    sent_id_count[base_id] = sent_id_count.get(base_id, 0) + 1
                    current_sent_id = f"{base_id}_{sent_id_count[base_id]}"
                continue  # skip all # lines
            
            # blank line indicates the end of a sentence
            if not line:
                current_sent_id = None
                continue

            columns = line.split("\t")
            # insert the unique sentence id as a new column at the beginning
            if current_sent_id is not None:
                columns.insert(0, current_sent_id)
            rows.append(columns)

    # the added sent_id plus the original 11 columns from the conllu file.
    col_names = [
        "sent_id", "word_id", "word", "lemma", "UPOS", "XPOS", "features", "head_distance", "dep", "adv_dep", "space_after", "predicate", "arguments"]
    # if for some reason the sent_id wasn't added, adjust column names accordingly
    if rows and len(rows[0]) == 12:
        col_names = col_names[1:]
        
    df = pd.DataFrame(rows, columns=col_names)
    return df

def convert_df(df):
    """
    Converts a DataFrame into a structured format for BERT-based Semantic Role Labeling (SRL).
    
    Parameters:
    df : pandas.DataFrame
        A DataFrame containing:
        - 'sent_id': Unique sentence identifier
        - 'word': Tokenized words in the sentence
        - 'arguments': e.g. "ARG0", "ARG1", "_" for non-arguments
        - 'predicate': Predicate annotations

    Returns:
    list of dict:
        A list where each dictionary represents a sentence with:
        - 'id': Sentence identifier
        - 'tokens': List of words in the sentence
        - 'labels': List of numerical labels for arguments
        - 'labels_word': List of argument labels in text format
        - 'predicate': Binary list (1 if token is a predicate, 0 otherwise)
        - 'pred_word': List of predicate annotations
    """

    # Process each group (sentence) using the global mapping.
    result = [
        {
            'id': str(i),
            'tokens': group['word'].tolist(),
            'labels': [label2id[item] for item in group['arguments']],
            'labels_word': ['_' if item == '_' else item for item in group['arguments']] ,
            'predicate': [0 if val == '_' else 1 for val in group['predicate']],
            'pred_word': group['predicate'].tolist()
        }
        for i, (_, group) in enumerate(df.groupby('sent_id', sort=False))
    ]
    
    return result

def augment_data(data):
    """
    Adds special tokens to mark predicates in the input data.

    Parameters:
    data : list of dict
        A list of sentence dictionaries containing:
        - 'tokens': List of words in the sentence
        - 'labels': List of integer labels for each token
        - 'predicate': Binary list indicating predicates (1 for predicates, 0 otherwise)
        - 'pred_word': List of predicate annotations

    Returns:
    list of dict
        A modified list where:
        - "[PRED]" is inserted before each predicate token
        - Labels for inserted tokens are set to -100 (ignored during training)
        - Other sentence structure remains unchanged
    """
        
    
    final =[]
    for sentence in data:
        
        tokens = sentence['tokens']
        predicate = sentence['predicate']
        pred_word = sentence['pred_word']
        labels = sentence['labels']
    
        # count the number of 1s in the predicates (1 indicates predicate)
        pred_count = predicate.count(1)
    
        # process tokens based on the predicates
        processed_tokens = []
        processed_labels = []
        processed_predicate = []
        processed_pred_word = []
        for idx, (token, pred) in enumerate(zip(tokens, predicate)):
            if pred == 1:
                processed_labels.append(-100) # if pred == 1 (the particular token is a predicate), it will be recognized as a special token
                processed_predicate.append(predicate[idx])
                processed_pred_word.append(pred_word[idx])
                processed_tokens.append("[PRED]")
                
            processed_tokens.append(token)
            processed_labels.append(labels[idx])
            processed_predicate.append(predicate[idx])
            processed_pred_word.append(pred_word[idx])
    
        final.append({
            'tokens':processed_tokens,
            'labels': processed_labels,
            'predicate': processed_predicate,
            'pred_word': processed_pred_word
        })
    return final

Preprocessing the data

Before we can feed those texts to our model, we need to preprocess them. This is done by a Transformers Tokenizer which will (as the name indicates) tokenize the inputs (including converting the tokens to their corresponding IDs in the pretrained vocabulary) and put it in a format the model expects, as well as generate the other inputs that model requires.

To do all of this, we instantiate our tokenizer with the AutoTokenizer.from_pretrained method, which will ensure:

  • we get a tokenizer that corresponds to the model architecture we want to use,
  • we download the vocabulary used when pretraining this specific checkpoint.

That vocabulary will be cached, so it's not downloaded again the next time we run the cell.

Se need to do some processing on our labels as the input ids returned by the tokenizer are longer than the lists of labels our dataset contain, first because some special tokens might be added (for example, [CLS] and a [SEP]) and then because of possible splits of words in multiple tokens.

Thankfully, the tokenizer returns outputs that have a word_ids method which can help us.

The method word_ids returns a list with the same number of elements as our processed input ids, mapping special tokens to None and all other tokens to their respective word. This way, we can align the labels with the processed input ids.

def tokenized_input(data):
    """
    Processes a list of sentences by tokenizing them and aligning their labels with the tokenized output.
    
    Parameter:
    data : list[dict]: the list of dictionaries converted and augmented.
        
    Make changes in each sentence dictionary in place by adding:
    - labels (list[int]): aligned labels where -100 represents tokens that are special tokens 
    - input_ids (list[int]): token ids that mark their place in the pretrained vocabulary
    - attention_mask (list[int]): 1 indicates that the token is real and should be attended to. 0 indicates that the token is padding 
    and should be ignored.
    - word_ids (list[int]): mapping from token positions to word positions
    - tokenized_tokens (list[str]): The subtokens after tokenization
            

    Notes:
        - Uses is_split_into_words=True as input is already tokenized
        - label value -100 is used for tokens that should be ignored in loss computation
    """
    tokenized_data=[]
    for sent in data:
        single_sent=sent.copy()  # copy the sentence to avoid modifying the original
        tokenized_input = tokenizer(sent["tokens"], is_split_into_words=True) # tokenize the sentence into subwords
        tokens = tokenizer.convert_ids_to_tokens(tokenized_input["input_ids"]) # convert subword token ids (vocabulary positions) to subword tokens
        word_ids = tokenized_input.word_ids() # subword word ids
        aligned_labels = [-100 if i is None else sent["labels"][i] for i in word_ids]
        
        single_sent["labels"] = aligned_labels
        single_sent["input_ids"] = tokenized_input["input_ids"] # subword token ids
        single_sent["attention_mask"] = tokenized_input["attention_mask"]
        single_sent['word_ids'] = word_ids # subword word ids
        single_sent['tokenized_tokens'] = tokens # subword tokens
        tokenized_data.append(single_sent)
    return tokenized_data

def preprocess_example(example):
    """
    Preprocesses input data by extracting relevant features for model input.
    
    This function takes a dictionary containing various features of a training example
    and returns a new dictionary with only the essential features needed for model training:
    input ids, attention mask, and labels.
    
    Argument:
        example (dict): a dictionary containing the example's features, including at minimum:
            - input_ids: ids marking the token's position in pretrained vocabulary
            - attention_mask: binary mask indicating valid input tokens
            - labels: aligned labels where -100 represents tokens that are special tokens
            
    Returns:
        dict: a preprocessed dictionary containing:
            - input_ids

def compute_metrics(p):
    """
    Compute evaluation metrics for token classification predictions.

    This function calculates precision, recall, f1 score, and accuracy for token-level
    predictions. It handles special tokens by removing ignored indices (-100) before computation.

    Argument:
        p (tuple): A tuple containing:
            - predictions (numpy.ndarray): model predictions with shape (batch_size, sequence_length, num_labels)
            - labels (numpy.ndarray): ground truth labels with shape (batch_size, sequence_length)
                Special tokens are marked with -100 and will be ignored in evaluation

    Returns:
        dict: A dictionary containing the following metrics:
            - precision (float): overall precision across all classes
            - recall (float): overall recall across all classes
            - f1 (float): overall f1 score across all classes
            - accuracy (float): overall accuracy across all classes

    Notes:
        - Predictions are converted from logits to label indices using argmax
        - Special tokens (labeled as -100) are removed before metric computation
    """
    predictions, labels = p
    predictions = np.argmax(predictions, axis=2)
    
    # Remove ignored index (special tokens)
    true_predictions = [
        [label_list[p] for (p, l) in zip(prediction, label) if l != -100]
        for prediction, label in zip(predictions, labels)
    ]
    true_labels = [
        [label_list[l] for (p, l) in zip(prediction, label) if l != -100]
        for prediction, label in zip(predictions, labels)
    ]

    results = metric.compute(predictions=true_predictions, references=true_labels)
    return {
        "precision": results["overall_precision"],
        "recall": results["overall_recall"],
        "f1": results["overall_f1"],
        "accuracy": results["overall_accuracy"],
    }

Answer: First, the raw dataframe, which contains columns of tokens, predicates, and arguments, is converted into a list of dictionaries, where each dictionary represents information of a sentence. Each dictionary contains fields including 'tokens', 'labels' (numerical labels for different classes, e.g. 0 for ARG0 and 1 for ARG1), 'predicates' (binary labels, 0 for not predicate and 1 for a predicate), 'labels_word' (labels in textual form) and 'pred_word' (the original predicate in string).

After that, an augmentation step is applied to the field 'tokens' and 'labels'. Tokens associated with predicates are marked with a special token [PRED]. This additional marking helps the model to refer to the context around predicates. The special token inserted during this step is ignored during loss calculations by setting its labels to -100.

The data is then tokenized using the AutoTokenizer. This tokenizer converts words into subwords, and provide input ids based on the model's pretrained vocabulary, and at the same time, predicate tokens are updated as special tokens. Subwords and their original tokens are mapped using the word_ids method of the tokenizer. Word ids are set as -100 for special tokens. A new list of dictionaries is created with fields including 'labels' (numerical labels for different argument classes), 'input_ids' (pretrained vocabulary ids), 'attention_mask' (a binary mask for distinguishing real tokens from padding), 'word_ids' (sequntial ids mapping subwords and their original tokens) and 'tokenized_tokens' (tokenized tokens).

def process_data(data_list):
    """
    Takes a list of tokenized sentences and converts their token-level
    predictions and gold labels into word-level predictions and gold labels using
    majority voting. It removes unwanted special tokens like [CLS], [SEP], and padding tokens.

    1. Filters out special tokens and their associated predictions/labels.
    2. Groups subword tokens belonging to the same original word based on 'word_ids'.
    3. Uses majority voting to assign a single prediction and gold label to each word.

    Argument:
        data_list (list[dict]): A list of dictionaries (augmented test data list), where each dictionary represents
                                a tokenized sentence. Each dictionary must include:
                                - 'word_ids': list mapping subwords to original words.
                                - 'predictions': token-level predictions from the model.
                                - 'gold': token-level gold (true) labels.
                                - 'tokenized_tokens': the tokenized tokens (with subwords).

    Returns:
        tuple: A tuple containing:
               - processed_predictions_list (list[list[int]]): word-level predictions for all sentences.
               - processed_gold_list (list[list[int]]): word-level gold labels for all sentences.
    """
    def process_single(data):
        word_ids = data['word_ids']
        predictions = data['predictions']
        gold = data['gold']
        tokenized_tokens = data['tokenized_tokens']
        # filter out special tokens like [CLS], [SEP], [PAD], etc
        word_ids = [b for a, b in zip(tokenized_tokens, word_ids) if a not in ['[CLS]','[SEP]','[PAD]','[PRED]','[UNK]','[MASK]']]
        predictions = [b for a, b in zip(tokenized_tokens, predictions) if a not in ['[CLS]','[SEP]','[PAD]','[PRED]','[UNK]','[MASK]']]
        gold = [b for a, b in zip(tokenized_tokens, gold) if a not in ['[CLS]','[SEP]','[PAD]','[PRED]','[UNK]','[MASK]']]
        
        processed_predictions = [] # word-level predictions
        processed_gold = []
        
        # track the current word group
        current_word_id = None
        current_predictions = []
        current_gold = []
        
        for idx, word_id in enumerate(word_ids):
            if word_id != current_word_id:
                # if a new word_id starts, calculate majority vote for the previous group
                if current_predictions:
                    # use majority voting to decide the final prediction/label for the word
                    processed_predictions.append(Counter(current_predictions).most_common(1)[0][0])
                    # most_common(1) returns the single most common element
                    # 1st [0] extracts the first tuple from the list of most common elements. e.g. [(1, 2)] becomes (1, 2)
                    # 2nd [0] extracts the first element of the tuple (the most common prediction itself). e.g. (1, 2) becomes 1
                    processed_gold.append(Counter(current_gold).most_common(1)[0][0])
                
                # reset for the new word group
                current_word_id = word_id
                current_predictions = [predictions[idx]]
                current_gold = [gold[idx]]
            else:
                # ddd to the current group
                current_predictions.append(predictions[idx])
                current_gold.append(gold[idx])
        
        # deal with the last word group
        if current_predictions:
            processed_predictions.append(Counter(current_predictions).most_common(1)[0][0])
            processed_gold.append(Counter(current_gold).most_common(1)[0][0])
        
        return processed_predictions, processed_gold

    # lists to hold results for all sentences
    processed_predictions_list = []
    processed_gold_list = []
    
    # process each sentence in the dataset
    for data in data_list:
        processed_predictions, processed_gold = process_single(data)
        processed_predictions_list.append(processed_predictions)
        processed_gold_list.append(processed_gold)
    
    return processed_predictions_list, processed_gold_list

Question: Explain and exemplify, providing 1 or 2 examples, how you process the output of the model. Make sure that your systems predictions match the tokenization required by the shared-task.

Answer: Since BERT tokenizes words into subwords, the predictions are first provided at the subword level, but the task requires word-level predictions. Therefore, processing of the predictions was conducted to ensure that subword predictions are mapped back to the original words.

For each tokenized sentence, the subword-level predictions and gold labels are extracted from the model's output. The word_ids maps subwords back to their corresponding words and special tokens like [CLS], [SEP], and padding are excluded. The predictions and gold labels for subwords belonging to the same word are grouped based on their word ids.

Within each word group, majority voting is applied to decide the final prediction and gold label for the word. For example, if a word split into tree subwords has predictions [2, 1, 2], the majority prediction 2 will be chosen as the final prediction for the word. This logic ensures a consistent word-level label and consider cases when the subword predictions differ.

For example, a tokenized sentence with the subwords and their corresponding predictions:

Subwords: ['red', '##ic', '##ulous']

Word ids: [6, 6, 6]

Predictions: [3, 5, 3]

The subwords 'red' '##ic' '##ulous' belong to the same word (word id: 6). The predictions for these subwords are [3, 5, 3]. Using majority voting, the final word-level prediction for 'rediculous' becomes 3.

This process ensures that the final predictions is in word level required by the task and filtering out irrelevant tokens.

def output_tsv(input_file, output_file, prediction, gold):
    """
    Writes predictions and gold labels to an output TSV file.

    Parameters:
    input_file : str
        Path to input TSV file.
    output_file : str
        Path to output TSV file.
    prediction : list of list of int
        Predicted labels.
    gold : list of list of int
        Gold labels.

    Returns: None
    """
    outfile = open(output_file, "w")

    index = 0
    for line in open(input_file, "r", encoding="utf-8"):
        line = line.rstrip("\n")

        if line.startswith("#"):  # preserve metadata lines (e.g., # sent_id, # text)
            outfile.write(line)
            continue

        if len(line.split()) > 0:
            columns = line.split("\t")

            pred_label = prediction[index]
            gold_label = gold[index]

            # append predicted and gold labels as new columns
            outfile.write(line + "\t" + str(pred_label) + "\t" + str(gold_label) + "\n")
            index += 1
        else:
            outfile.write("\n")

    outfile.close()

def compute_metrics(p):
    """
    Compute evaluation metrics for token classification predictions.

    This function calculates precision, recall, f1 score, and accuracy for token-level
    predictions. It handles special tokens by removing ignored indices (-100) before computation.

    Argument:
        p (tuple): A tuple containing:
            - predictions (numpy.ndarray): model predictions with shape (batch_size, sequence_length, num_labels)
            - labels (numpy.ndarray): ground truth labels with shape (batch_size, sequence_length)
                Special tokens are marked with -100 and will be ignored in evaluation

    Returns:
        dict: A dictionary containing the following metrics:
            - precision (float): overall precision across all classes
            - recall (float): overall recall across all classes
            - f1 (float): overall f1 score across all classes
            - accuracy (float): overall accuracy across all classes

    Notes:
        - The function expects label_list to be defined in the outer scope
        - The metric object should be defined in the outer scope and support the
          compute() method with predictions and references parameters
        - Predictions are converted from logits to label indices using argmax
        - Special tokens (labeled as -100) are removed before metric computation
    """
    predictions, labels = p
    predictions = np.argmax(predictions, axis=2)
    
    # Remove ignored index (special tokens)
    true_predictions = [
        [label_list[p] for (p, l) in zip(prediction, label) if l != -100]
        for prediction, label in zip(predictions, labels)
    ]
    true_labels = [
        [label_list[l] for (p, l) in zip(prediction, label) if l != -100]
        for prediction, label in zip(predictions, labels)
    ]

    results = metric.compute(predictions=true_predictions, references=true_labels)
    return {
        "precision": results["overall_precision"],
        "recall": results["overall_recall"],
        "f1": results["overall_f1"],
        "accuracy": results["overall_accuracy"],
    }

def mega_function(sentence_data, saved_tokenizer):
    """
    This is a function combining key components of functions including convert_df, augment_data, tokenized_input, preprocess_example, make_dataset.
    Specially designed for semantic role labeling using trained BERT model on standalone sentences.

    Parameters:
    sentence_data : pandas.DataFrame
        containing sentence-level data with the following columns:
        - 'word': tokens in the sentence.
        - 'predicate': binary integer indicators for predicates (1 means the token is a predicate, 0 means the token is not a predicate).
    saved_tokenizer: tokenizer added with special tokens

    Returns:
    tuple:
        - tokenized_data: list of dict
            A list of dictionaries where each dictionary contains the following keys:
            - 'input_ids'
            - 'attention_mask'
            - 'word_ids'
            - 'tokenized_tokens'
        - final_dataset: datasets.Dataset
            A Hugging Face Dataset object containing the tokenized input data for model training, containing:
            - 'input_ids'
            - 'attention_mask'
    """
    x = [
            {
                # 'id': '1',
                'tokens': sentence_data['word'].tolist(),
                'predicate': [0 if val == '_' else 1 for val in sentence_data['predicate']]
                # 'pred_word': sentence_data['pred_word'].tolist()
            }
        ]
    
    y = []
    for sentence in x:
        
        tokens = sentence['tokens']
        predicate = sentence['predicate']
        # pred_word = sentence['pred_word']
    
        # Process tokens based on the predicates
        processed_tokens = []
        processed_labels = []
        processed_predicate = []
        # processed_pred_word = []
        for idx, (token, pred) in enumerate(zip(tokens, predicate)):
            if pred == 1:
                processed_labels.append(-100) # if pred == 1, it will be recognized as a special token
                processed_predicate.append(predicate[idx])
                # processed_pred_word.append(pred_word[idx])
                processed_tokens.append("[PRED]")
                
            processed_tokens.append(token)
            processed_labels.append(labels[idx])
            processed_predicate.append(predicate[idx])
            # processed_pred_word.append(pred_word[idx])
    
        y.append({
            'tokens':processed_tokens,
            'labels': processed_labels
            # 'pred_word': processed_pred_word
        })

    tokenized_data=[]
    for sent in y:
        single_sent={}
        tokenized_input = saved_tokenizer(sent["tokens"], is_split_into_words=True)
        tokens = saved_tokenizer.convert_ids_to_tokens(tokenized_input["input_ids"])
        word_ids = tokenized_input.word_ids()
        
        single_sent["input_ids"] = tokenized_input["input_ids"]
        single_sent["attention_mask"] = tokenized_input["attention_mask"]
        single_sent['word_ids'] = word_ids
        single_sent['tokenized_tokens'] = tokens
        tokenized_data.append(single_sent)

    
    processed_data = []
    for example in tokenized_data:
        processed_data.append({
            "input_ids": tokenized_data[0]["input_ids"],
            "attention_mask": tokenized_data[0]["attention_mask"],
        })

        
    final_dataset = Dataset.from_dict({
        'input_ids': [example['input_ids'] for example in processed_data],
        'attention_mask': [example['attention_mask'] for example in processed_data]
    })
    
    return tokenized_data, final_dataset
